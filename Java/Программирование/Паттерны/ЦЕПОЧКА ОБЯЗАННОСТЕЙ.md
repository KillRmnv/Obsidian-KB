[[Поведенческий паттерн|Поведенческий паттерн проектирования]], который позволяет передать запросы последовательно по цепочке обработчиков.
***
## Пример

```python
from abc import ABC, abstractmethod
from typing import Optional

# Базовый класс обработчика
class Handler(ABC):
    def __init__(self):
        self._next_handler: Optional[Handler] = None

    def set_next(self, handler: 'Handler') -> 'Handler':
        self._next_handler = handler
        return handler

    @abstractmethod
    def handle(self, request: dict) -> Optional[str]:
        if self._next_handler:
            return self._next_handler.handle(request)
        return None

# Конкретные обработчики
class AuthenticationHandler(Handler):
    def handle(self, request: dict) -> Optional[str]:
        if not request.get('authenticated'):
            return "Ошибка: Пользователь не аутентифицирован"
        print("Аутентификация пройдена")
        return super().handle(request)

class AuthorizationHandler(Handler):
    def handle(self, request: dict) -> Optional[str]:
        if not request.get('has_permission'):
            return "Ошибка: Недостаточно прав"
        print("Авторизация пройдена")
        return super().handle(request)

class ValidationHandler(Handler):
    def handle(self, request: dict) -> Optional[str]:
        if not request.get('is_valid'):
            return "Ошибка: Невалидные данные"
        print("Валидация пройдена")
        return super().handle(request)

class ProcessingHandler(Handler):
    def handle(self, request: dict) -> Optional[str]:
        print("Обработка запроса завершена успешно!")
        return "Успех: Запрос обработан"

# Создаем цепочку обработчиков
auth_handler = AuthenticationHandler()
auth_handler.set_next(
    AuthorizationHandler()
).set_next(
    ValidationHandler()
).set_next(
    ProcessingHandler()
)
```
