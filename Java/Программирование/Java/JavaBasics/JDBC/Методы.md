# JDBC: Классы и методы для работы с БД

## Основные интерфейсы JDBC

### 1. **Statement**
Используется для выполнения простых SQL-запросов без параметров.

```java
// Создание
Statement stmt = connection.createStatement();

// Основные методы:
// Выполнение SELECT
ResultSet rs = stmt.executeQuery("SELECT * FROM users");

// Выполнение INSERT/UPDATE/DELETE
int rowsAffected = stmt.executeUpdate("UPDATE users SET name='John' WHERE id=1");

// Выполнение любых SQL команд
boolean isResultSet = stmt.execute("DROP TABLE users");

// Пакетная обработка
stmt.addBatch("INSERT INTO users VALUES (1, 'John')");
stmt.addBatch("INSERT INTO users VALUES (2, 'Jane')");
int[] results = stmt.executeBatch();
```

### 2. **PreparedStatement** (Рекомендуемый)
Предотвращает SQL-инъекции, предварительная компиляция, работа с параметрами.

```java
// Создание с параметрами (обозначаются ?)
PreparedStatement pstmt = connection.prepareStatement(
    "INSERT INTO users (name, age, email) VALUES (?, ?, ?)"
);

// Установка параметров по индексу (начиная с 1)
pstmt.setString(1, "John Doe");     // для VARCHAR/CHAR
pstmt.setInt(2, 25);                 // для INTEGER
pstmt.setString(3, "john@email.com");

// Другие методы set:
pstmt.setLong(1, 100L);
pstmt.setDouble(2, 99.99);
pstmt.setBoolean(3, true);
pstmt.setDate(4, java.sql.Date.valueOf("2023-01-01"));
pstmt.setTimestamp(5, new Timestamp(System.currentTimeMillis()));
pstmt.setObject(6, myObject);        // для сложных типов
pstmt.setNull(7, Types.VARCHAR);     // установка NULL

// Выполнение
ResultSet rs = pstmt.executeQuery();     // для SELECT
int rows = pstmt.executeUpdate();         // для INSERT/UPDATE/DELETE

// Пример с SELECT и параметрами
PreparedStatement pstmt = conn.prepareStatement(
    "SELECT * FROM users WHERE age > ? AND city = ?"
);
pstmt.setInt(1, 18);
pstmt.setString(2, "Moscow");
ResultSet rs = pstmt.executeQuery();
```

### 3. **CallableStatement**
Для вызова хранимых процедур и функций.

```java
// Процедура без параметров
CallableStatement cstmt = conn.prepareCall("{call get_all_users()}");

// Процедура с параметрами
CallableStatement cstmt = conn.prepareCall("{call add_user(?, ?, ?)}");
cstmt.setString(1, "Alice");
cstmt.setInt(2, 30);
cstmt.setString(3, "alice@mail.com");
cstmt.execute();

// Процедура с выходными параметрами
CallableStatement cstmt = conn.prepareCall("{call get_user_count(?)}");
cstmt.registerOutParameter(1, Types.INTEGER);
cstmt.execute();
int count = cstmt.getInt(1);

// Функция с возвращаемым значением
CallableStatement cstmt = conn.prepareCall("{? = calculate_bonus(?, ?)}");
cstmt.registerOutParameter(1, Types.DECIMAL);
cstmt.setInt(2, 1000);
cstmt.setInt(3, 5);
cstmt.execute();
BigDecimal bonus = cstmt.getBigDecimal(1);
```

### 4. **ResultSet**
Для обработки результатов запроса.

```java
Statement stmt = conn.createStatement();
ResultSet rs = stmt.executeQuery("SELECT id, name, age FROM users");

// Навигация по результатам
while (rs.next()) {                    // перемещение на следующую строку
    int id = rs.getInt("id");           // по имени колонки
    String name = rs.getString("name");
    int age = rs.getInt("age");         // по индексу (начиная с 1)
    
    // Альтернативный способ - по индексу
    // int id = rs.getInt(1);
}

// Другие методы навигации:
rs.beforeFirst();    // перемещение перед первой записью
rs.first();          // на первую запись
rs.last();           // на последнюю запись
rs.absolute(5);      // на 5-ю запись
rs.relative(-2);     // на 2 записи назад
rs.previous();       // на предыдущую запись

// Получение метаданных
ResultSetMetaData meta = rs.getMetaData();
int columnCount = meta.getColumnCount();
String columnName = meta.getColumnName(1);
String columnType = meta.getColumnTypeName(1);
```

### 5. **Connection**
Управление соединением с БД.

```java
// Получение соединения
Connection conn = DriverManager.getConnection(
    "jdbc:postgresql://localhost:5432/mydb",
    "user", 
    "password"
);

// Настройки
conn.setAutoCommit(false);              // отключение авто-коммита
conn.setTransactionIsolation(Connection.TRANSACTION_READ_COMMITTED);

// Управление транзакциями
conn.commit();                           // подтверждение транзакции
conn.rollback();                         // откат транзакции
conn.rollback(savepoint);                 // откат к точке сохранения

// Создание savepoint
Savepoint sp = conn.setSavepoint("point1");

// Закрытие
conn.close();
```

## Полный пример работы

```java
public class UserDAO {
    private Connection connection;
    
    public UserDAO(Connection connection) {
        this.connection = connection;
    }
    
    // Вставка с возвратом сгенерированного ключа
    public User create(User user) throws SQLException {
        String sql = "INSERT INTO users (name, email) VALUES (?, ?)";
        
        try (PreparedStatement pstmt = connection.prepareStatement(
                sql, Statement.RETURN_GENERATED_KEYS)) {
            
            pstmt.setString(1, user.getName());
            pstmt.setString(2, user.getEmail());
            
            int affectedRows = pstmt.executeUpdate();
            
            if (affectedRows == 0) {
                throw new SQLException("Creating user failed");
            }
            
            // Получение сгенерированного ID
            try (ResultSet generatedKeys = pstmt.getGeneratedKeys()) {
                if (generatedKeys.next()) {
                    user.setId(generatedKeys.getLong(1));
                }
            }
            
            return user;
        }
    }
    
    // Поиск с параметрами
    public List<User> findByAgeRange(int minAge, int maxAge) throws SQLException {
        String sql = "SELECT * FROM users WHERE age BETWEEN ? AND ?";
        List<User> users = new ArrayList<>();
        
        try (PreparedStatement pstmt = connection.prepareStatement(sql)) {
            pstmt.setInt(1, minAge);
            pstmt.setInt(2, maxAge);
            
            try (ResultSet rs = pstmt.executeQuery()) {
                while (rs.next()) {
                    users.add(mapRow(rs));
                }
            }
        }
        
        return users;
    }
    
    // Пакетная вставка
    public void batchInsert(List<User> users) throws SQLException {
        String sql = "INSERT INTO users (name, email) VALUES (?, ?)";
        
        try (PreparedStatement pstmt = connection.prepareStatement(sql)) {
            for (User user : users) {
                pstmt.setString(1, user.getName());
                pstmt.setString(2, user.getEmail());
                pstmt.addBatch();
            }
            
            int[] results = pstmt.executeBatch();
        }
    }
    
    private User mapRow(ResultSet rs) throws SQLException {
        User user = new User();
        user.setId(rs.getLong("id"));
        user.setName(rs.getString("name"));
        user.setEmail(rs.getString("email"));
        user.setAge(rs.getInt("age"));
        return user;
    }
}
```
